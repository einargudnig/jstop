**Key decisions for this plan:**
- lsof -i TCP -n -P -a -p {pid} — filters to TCP, numeric addresses/ports, only for this PID; faster than full lsof
- Parse LISTEN lines only — ignore ESTABLISHED (outbound connections); only care about servers
- Store ports as [Int] on JSProcess — allows multiple ports (e.g. Vite + HMR websocket)
- Show lowest port in badge — most dev servers listen on one port; lowest is the HTTP one
- Open first (lowest) port in browser — http://localhost:{port}
- lsof called per-process during fetchProcesses() — acceptable for <10 processes; not a bottleneck
- NSWorkspace.shared.open() — correct macOS API for opening URLs in default browser
- No sandbox required — lsof runs freely since App Sandbox is disabled

**2. Update `NodeProcesses/ProcessManager.swift` — add detectPorts and call it from fetchProcesses:**

Add the following method to the `ProcessManager` class (after `kill(pid:)`):

```swift
func detectPorts(for pid: Int) -> [Int] {
    let task = Process()
    task.executableURL = URL(fileURLWithPath: "/usr/sbin/lsof")
    // -i TCP: TCP sockets only
    // -n: no hostname resolution (faster)
    // -P: no port name resolution (show numbers, not "http")
    // -a: AND the following filters
    // -p {pid}: only this process
    task.arguments = ["-i", "TCP", "-n", "-P", "-a", "-p", "\(pid)"]

    let outPipe = Pipe()
    let errPipe = Pipe()
    task.standardOutput = outPipe
    task.standardError = errPipe

    do {
        try task.run()
        task.waitUntilExit()
    } catch {
        return []
    }

    let data = outPipe.fileHandleForReading.readDataToEndOfFile()
    guard let output = String(data: data, encoding: .utf8) else { return [] }

    // Parse lines looking for LISTEN state
    // Example line: node  1234  user  22u  IPv6  0t0  TCP *:3000 (LISTEN)
    var ports: [Int] = []
    for line in output.components(separatedBy: "\n") {
        guard line.contains("(LISTEN)") else { continue }
        // Port is in the address field — find last ":" before " (LISTEN)"
        // Handles both "*:3000" and "127.0.0.1:3000" formats
        let components = line.components(separatedBy: .whitespaces).filter { !$0.isEmpty }
        // The address is typically the 9th field (index 8)
        // Find the component that ends before "(LISTEN)" — it contains the address
        if let addrComponent = components.first(where: { $0.contains(":") && !$0.hasPrefix("IPv") }),
           let portStr = addrComponent.split(separator: ":").last,
           let port = Int(portStr) {
            if !ports.contains(port) {
                ports.append(port)
            }
        }
    }
    return ports.sorted()
}
```

Then update `fetchProcesses()` — after building the `found` array, populate ports. Replace the final line `self.processes = found` with:

```swift
import SwiftUI

struct ProcessRowView: View {
    let process: JSProcess
    @ObservedObject var manager: ProcessManager

    var body: some View {
        HStack(alignment: .top, spacing: 8) {
            VStack(alignment: .leading, spacing: 2) {
                HStack(spacing: 6) {
                    Text(process.name)
                        .font(.system(.body, design: .monospaced))
                        .fontWeight(.semibold)
                    Text("PID \(process.pid)")
                        .font(.caption)
                        .foregroundColor(.secondary)
                        .padding(.horizontal, 5)
                        .padding(.vertical, 1)
                        .background(Color.secondary.opacity(0.12))
                        .cornerRadius(3)

                    // Port badge — shows lowest detected port
                    if let port = process.listeningPorts.first {
                        Text(":\(port)")
                            .font(.caption)
                            .foregroundColor(.blue)
                            .padding(.horizontal, 5)
                            .padding(.vertical, 1)
                            .background(Color.blue.opacity(0.10))
                            .cornerRadius(3)
                    }
                }
                Text(process.args)
                    .font(.caption)
                    .foregroundColor(.secondary)
                    .lineLimit(2)
                    .truncationMode(.middle)
            }
            Spacer(minLength: 4)

            // Open in browser button — only shown when a port is detected
            if let port = process.listeningPorts.first {
                Button {
                    if let url = URL(string: "http://localhost:\(port)") {
                        NSWorkspace.shared.open(url)
                    }
                } label: {
                    Image(systemName: "safari")
                        .font(.system(size: 14))
                        .foregroundColor(.blue.opacity(0.75))
                }
                .buttonStyle(.plain)
                .help("Open http://localhost:\(port) in browser")
            }

            // Kill button
            Button {
                manager.kill(pid: process.pid)
            } label: {
                Image(systemName: "xmark.circle.fill")
                    .font(.system(size: 16))
                    .foregroundColor(.red.opacity(0.75))
            }
            .buttonStyle(.plain)
            .help("Kill process \(process.pid)")
        }
        .padding(.horizontal, 12)
        .padding(.vertical, 8)
    }
}
```

Notes:
- `process.listeningPorts.first` — shows lowest port (sorted in detectPorts). If empty, no badge or Open button.
- `NSWorkspace.shared.open()` — correct macOS API; opens URL in default browser. Requires `import SwiftUI` (NSWorkspace is available via AppKit which SwiftUI imports on macOS).
- Safari icon (`safari` SF Symbol) — immediately recognizable as "open in browser"
- Kill button stays — the full row now has: name/pid/port badge / args / [Open button] / kill button
